             /**
     * Listar cards do dashboard
     */
    public function listar()
    {
        $cards = [
            [
                "titulo"     => "Produtos",
                "quantidade" => count(ProdutoDao::listarAtivos())
            ],
            [
                "titulo"     => "Categorias",
                "quantidade" => count(\App\Dao\Categoria\CategoriaDao::listarAtivas())
            ],
            [
                "titulo"     => "Produtos em Destaque",
                "quantidade" => count(ProdutoDestaqueDao::listarParaPainel())
            ],
        ];

        self::Mensagemjson(
            "Dashboard carregado com sucesso",
            200,
            $cards
        );
    }

    /**
     * Buscar produto pelo slug
     */
    public function produtoPorSlug(string $slug)
    {
        $produto = ProdutoDao::buscarPorSlug($slug);

        if (!$produto) {
            return self::Mensagemjson(
                "Produto n√£o encontrado",
                404
            );
        }

        return self::Mensagemjson(
            "Produto encontrado",
            200,
            $produto
        );
    }
